library(ggplot2)
data("Batting")
Batting %>% group_by(yearID) %>% 
  summarize(BABIP = (sum(H, na.rm = T) - sum(HR, na.rm = T))/
              (sum(AB, na.rm = T) - sum(SO, na.rm = T) - sum(HR, na.rm = T) + sum(SF, na.rm = T))) %>% 
  filter(yearID>1953) %>% 
  ggplot(aes(x = yearID, y = BABIP)) + geom_point()
